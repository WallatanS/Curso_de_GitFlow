//////Lista de comandos básicos do Git

  git init (criar repositório)
  
  git add .
  git commit -m “texto informativo” 
  git commit -am "teste" //faz o add e logo sobre o commit
  git push //Envia os dados para o servidor
  git pull //Puxa os dados da servidor
  
  git checkout -b name-feature //cria uma branch e define ela como foco
 
  git branch develop //cria uma branch
  git branch -d nome-branch //remove uma branch
  git branch -m main //renomeia uma branch
  git push meu-repo --delete minha-branch
  git checkout name-feature

  git log //Ver histórico de commit
  git revert 0df45df //desfaz um commit
  git stash //salva as alterações de forma temporária
